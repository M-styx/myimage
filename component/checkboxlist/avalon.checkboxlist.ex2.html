配置check来设置checkboxlist初始选中状态

onSelect获取values: {{el}}  

<!DOCTYPE html>
<html>

    <head>
        <title>checkboxlist demo 2</title>
        <meta charset="UTF-8">
        <meta name="viewport" content="width=device-width">
        <link type="text/css" rel="stylesheet" href="../style/avalon.doc.css"
        />
        <script src="../../avalon.js"></script>
        <script src="../style/highlight/shCore.js"></script>
        <script>
            avalon.library("aoyou", {
                $ready: function () {
                    avalon.log("控件已经构建完毕")
                }
            })
            require(["./component/checkboxlist/avalon.checkboxlist"], function(avalon) {
                var demoVM = avalon.define({
                    $id: "demo",
                    checkedInputs: [],
                    $opts: {
                        val: ['w1', 'w2'],
                        data: [
                            { text : '文字1' , value : 'w1' } ,
                            { text : '文字2' , value : 'w2' } ,
                            { text : '文字3' , value : 'w3' } ,
                            { text : '文字4' , value : 'w4' }
                        ],
                        onSelect: function(e, inputsData){
                            demoVM.checkedInputs = inputsData
                        }
                    }
                })
                avalon.scan()
            });
        </script>
    </head>

    <body>
        <div class="wrapper">
            <div class="demo-show" ms-controller="demo">
                <fieldset>
                    <legend>配置val来设置checkboxlist初始选中状态</legend>
                    <aoyou:checkboxlist config="$opts" alltext=""></aoyou:checkboxlist>
                    <p>onSelect获取values:
                        <span ms-repeat="checkedInputs">{{el}} </span>
                    </p>
                </fieldset>
            </div>
            <div ms-skip></div>
        </div>
    </body>

</html>
