checkboxlist默认配置，利用onSelect回调获取组件状态

刚刚点击的选项index是: {{checkedIndex}}

onSelect获取values: {{el}}  

<!DOCTYPE html>
<html>
    
    <head>
        <title>checkboxlist demo 1</title>
        <meta charset="UTF-8">
        <meta name="viewport" content="width=device-width">
        <link type="text/css" rel="stylesheet" href="../style/avalon.doc.css"
        />
        <script src="../../avalon.js"></script>
        <script src="../style/highlight/shCore.js"></script>
        <script>
            avalon.library("aoyou", {
                $ready: function () {
                    avalon.log("控件已经构建完毕")
                }
            })
            require(["./component/checkboxlist/avalon.checkboxlist"], function () {
                var demoVM = avalon.define({
                    $id: "demo",
                    checkedInputs: [],
                    checkedIndex: "",
                    $opts: {
                        data: [
                            { text : '文字1' , value : 'w1' } ,
                            { text : '文字2' , value : 'w2' } ,
                            { text : '文字3' , value : 'w3' } ,
                            { text : '文字4' , value : 'w4' }
                        ],
                        onSelect: function(e, checkedData, index){
                            demoVM.checkedInputs = checkedData
                            demoVM.checkedIndex = index
                        }
                    }
                })
                avalon.scan()
            })
        </script>
    </head>
    
    <body ms-controller="demo">
        <fieldset>
            <legend>checkboxlist默认配置，利用onSelect回调获取组件状态</legend>
            <aoyou:checkboxlist config="$opts"></aoyou:checkboxlist>
            <p>刚刚点击的选项index是: {{checkedIndex}}</p>
            <p>onSelect获取values:
                <span ms-repeat="checkedInputs">{{el}} </span>
            </p>
        </fieldset>
        <div ms-skip></div>
    </body>

</html>
