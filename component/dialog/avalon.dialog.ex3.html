取消“close”按钮的dialog

说明： 此demo基本使用组件默认配置，只是通过showClose的配置使得组件不显示“关闭”按钮

取消“close”按钮的dialog 不显示关闭按钮 
<!DOCTYPE html>
<html>
    <head>
        <meta http-equiv="content-type" content="text/html;charset=utf-8" />
        <meta http-equiv="x-ua-compatible" content="ie=edge" />
        <title>dialog demo 3</title>
        <link type="text/css" rel="stylesheet"  href="../highlight/shCore.css"/>
        <link type="text/css" rel="stylesheet" href="../style/avalon.doc.css"/>
        <script src="../highlight/shCore.js"></script>
        <script src="../../avalon.js"></script>
    </head>
    <body>
        <div class="wrapper">

            <div class="demo-show" ms-controller="demo">
                <fieldset>  
                    <legend>取消“close”按钮的dialog</legend>
                    <button ms-click="show('aa')">打开对话框</button>
                    <aoyou:dialog $id="aa" $slot="content" show-close="false" >
                        不显示关闭按钮
                    </aoyou:dialog>
                </fieldset>
                <script>
                    require(["./component/dialog/avalon.dialog"], function () {
                        avalon.define({
                            $id: "demo",
                            show: function (id) {
                                avalon.vmodels[id].toggle = true;
                            },
                            title: "取消“close”按钮的dialog"
                        })
                        avalon.scan();
                    })
                </script>
            </div>
        </div>
    </body>
</html>
